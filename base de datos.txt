CREATE TABLE categoria (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

CREATE TABLE solucion_ia (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion TEXT NOT NULL,
    categoria_id INT,
    enlace VARCHAR(255),
    FOREIGN KEY (categoria_id) REFERENCES categoria(id)
);
